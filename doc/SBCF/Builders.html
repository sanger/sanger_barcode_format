def from_user_input(input)
  case input.to_s
  when HUMAN_BARCODE_FORMAT
    SangerBarcode.new(human_barcode: input)
  when MACHINE_BARCODE_FORMAT
    SangerBarcode.new(machine_barcode: input)
  else
    SangerBarcode.new(human_barcode: '')
  end
end
